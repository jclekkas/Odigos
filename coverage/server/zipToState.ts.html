/**
 * Derives a 2-letter state code from the first 3 digits of a US ZIP code.
 * Returns null if zip is undefined, shorter than 3 chars, or prefix not found.
 * ZIP itself is never stored — only the derived state code is persisted.
 */
export function zipToStateCode(zip: string | undefined): string | null {
  if (!zip || zip.length < 3) return null;
  return ZIP_PREFIX_TO_STATE[zip.substring(0, 3)] ?? null;
}
